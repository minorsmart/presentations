Financial Model


                      t <- 0
                      sales <- 1000000
                      varcosts <- sales * 0.7
                      fixcosts <- 50000
                      gresult <- sales - varcosts - fixcosts
                      taxes <- gresult * 0.25
                      nresult <- gresult - taxes

                      finres <- data.frame(t, sales, varcosts,
                                           fixcosts, gresult,
                                           taxes, nresult)

GRowth


                      growthcalc <- function(startvalue, rate) {
  
                                               endvalue <- startvalue * (1 + rate)
                                               return(endvalue)
  
                                    }

GRowth


                      growth <- 0.15
                      inflation <- 0.012
                      
                      for (s in c(1:10)) {
  
                      t <- c(t, s)
                      sales <- c(sales,
                                 growthcalc(sales[s],
                                            growth)
                                 )
                      fixcosts <- c(fixcosts,
                                    growthcalc(fixcosts[s],
                                               inflation)
                                    )
  
                      }

Risk


                      stdevg <- 0.12
                      stdevi <- 0.003

                      for (s in c(1:10)) {
  
                          t <- c(t, s)
                          sales <- c(sales,
                                     growthcalc(sales[s],
                                                rnorm(1, mean = growth, sd = stdevg)
                                                )
                                      )
                          
                          fixcosts <- c(fixcosts,
                                        growthcalc(fixcosts[s],
